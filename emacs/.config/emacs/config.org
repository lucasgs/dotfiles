#+title: Emacs Config

My literal Emacs configuration. This file is loaded by =init.el=
via =org-babel-load-file=. Edit this file and restart Emacs (or run
=org-babel-tangle= + reload) to apply changes.

* Bootstrap

Separate customizations written by Emacs into its own file so they
don't clutter this config.

#+begin_src emacs-lisp
(setq custom-file (locate-user-emacs-file "custom.el"))
(load custom-file t)
#+end_src

Keep backups in a dedicated directory instead of scattered next to the
original files.

#+begin_src emacs-lisp
(setq backup-directory-alist '((".*" . "~/.config/emacs/backups/")))
#+end_src

* UI

Strip away the chrome we don't need.

#+begin_src emacs-lisp
(scroll-bar-mode -1)             ; Disable ugly scrollbars
(tool-bar-mode -1)               ; Disable the graphical toolbar
(tooltip-mode -1)                ; Disable tooltips
(set-fringe-mode 10)             ; Give some breathing room on the sides
(menu-bar-mode -1)               ; Disable the menu bar
#+end_src

Show line numbers and the current column in the modeline.

#+begin_src emacs-lisp
(global-display-line-numbers-mode t) ; Show line numbers
(column-number-mode t)                ; Show column number in the modeline
#+end_src

Disable line numbers for modes where they don't make sense.

#+begin_src emacs-lisp
(dolist (mode '(org-mode-hook
                term-mode-hook
                eshell-mode-hook))
  (add-hook mode (lambda () (display-line-numbers-mode 0))))
#+end_src

* Company

A clean, modern inline auto-completion popup menu.

#+begin_src emacs-lisp
(use-package company
  :defer 2
  :diminish
  :custom
  (company-begin-commands '(self-insert-command))
  (company-idle-delay .1)
  (company-minimum-prefix-length 2)
  (company-show-numbers t)
  (company-tooltip-align-annotations 't)
  (global-company-mode t))

(use-package company-box
  :after company
  :diminish
  :hook (company-mode . company-box-mode))
#+end_src

* Keybindings

Open the init file from anywhere.

#+begin_src emacs-lisp
(defun lg/open-init-file ()
  "Open the literal Emacs configuration file config.org."
  (interactive)
  (find-file (expand-file-name "config.org" user-emacs-directory)))

** General

#+begin_src emacs-lisp
(use-package general
  :config
  (general-evil-setup)

  ;; set up 'SPC' as the global leader key
  (general-create-definer dt/leader-keys
    :states '(normal insert visual emacs)
    :keymaps 'override
    :prefix "SPC" ;; set leader
    :global-prefix "M-SPC") ;; access leader in insert mode

  (dt/leader-keys
    "." '(find-file :wk "Find file"))

  (dt/leader-keys
    "b" '(:ignore t :wk "Bookmarks/Buffers")
    "b b" '(switch-to-buffer :wk "Switch to buffer")
    "b i" '(ibuffer :wk "Ibuffer"))

  (dt/leader-keys
    "e" '(:ignore t :wk "Ediff/Eshell/Eval/EWW")
    "e b" '(eval-buffer :wk "Evaluate elisp in buffer")
    "e e" '(eval-expression :wk "Evaluate and elisp expression"))

  (dt/leader-keys
    "f" '(:ignore t :wk "Files")
    "f c" '((lambda () (interactive)
              (lg/open-init-file))
            :wk "Open config.org")))
#+end_src
